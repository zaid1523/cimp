ADVANCE C PROGRAMMING


========== 1. No Arguments, No Return Value ==========

#include <stdio.h>

void sum() {
    int a, b;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    printf("Sum = %d", a + b);
}

int main() {
    sum();
    return 0;
}

OUTPUT:
Enter two numbers: 10 20
Sum = 30


========== 2. Arguments, No Return Value ==========

#include <stdio.h>

void sum(int a, int b) {
    printf("Sum = %d", a + b);
}

int main() {
    int x, y;
    printf("Enter 2 numbers: ");
    scanf("%d %d", &x, &y);
    sum(x, y);
    return 0;
}

OUTPUT:
Enter 2 numbers: 45 25
Sum = 70


========== 3. No Arguments, With Return Value ==========

#include <stdio.h>

int sum() {
    int a, b;
    printf("Enter two numbers: ");
    scanf("%d %d", &a, &b);
    return (a + b);
}

int main() {
    int result = sum();
    printf("Sum = %d", result);
    return 0;
}

OUTPUT:
Enter two number: 78 69
Sum = 147


========== 4. Arguments, With Return Value ==========

#include <stdio.h>

int sum(int a, int b) {
    return (a + b);
}

int main() {
    int x, y, result;
    printf("Enter two numbers: ");
    scanf("%d %d", &x, &y);
    result = sum(x, y);
    printf("Sum = %d", result);
    return 0;
}

OUTPUT:
Enter two numbers: 69 53
Sum = 122
